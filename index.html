i++) { if (lines[i].length < 4) continue; let row = {}; for (let j = 0; j < headers.length; j++) { row[headers[j].trim()] = lines[i][j].trim(); } data.push(row); } return data; } // 從 Google Sheets 獲取數據並渲染 async function fetchSheetData() { try { // 將「立即更新」按鈕文字暫時改為「正在同步...」，並禁用按鈕 const button = document.querySelector('button[onclick="fetchSheetData()"]'); // 建立數據快速查找索引 let db = {}; data.forEach(item => { if(!db[item.產品]) db[item.產品] = {}; db[item.產品][item.金行] = { sell: parseInt(item.賣出價) || 0, buy: parseInt(item.買入價) || 0 }; }); // 遍歷每種產品找出最高最划算 products.forEach(p => { let sellValues = shops.map(s => db[p]?.[s]?.sell || Infinity);
